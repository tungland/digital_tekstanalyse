1.8.3. Ngram#

Ngram(words=["likestilling"], doctype="digavis", from_year=2010, to_year=2020).plot(lw = 5)

at = NgramNews(
    words=["likestilling"],
    title="aftenposten",
    from_year=1980,
    to_year=2000
    )

bt = NgramNews(
    words=["likestilling"],
    title="bergenstidende",
    from_year=1980,
    to_year=2000
    )

(at.ngram / bt.ngram).plot(lw=5)
